interface DiffLine {
  left: string;
  right: string;
  type: 'same' | 'added' | 'removed' | 'changed';
}
 
function computeDiff(left: string[], right: string[]): DiffLine[] {
  // LCS-based diff
  const m = left.length, n = right.length;
  const dp: number[][] = Array.from({ length: m + 1 }, () => new Array(n + 1).fill(0));
 
  for (let i = 1; i <= m; i++) {
    for (let j = 1; j <= n; j++) {
      dp[i][j] = left[i - 1] === right[j - 1] ? dp[i - 1][j - 1] + 1 : Math.max(dp[i - 1][j], dp[i][j - 1]);
    }
  }
 
  // Backtrack
  const result: DiffLine[] = [];
  let i = m, j = n;
  const temp: DiffLine[] = [];
  while (i > 0 || j > 0) {
    if (i > 0 && j > 0 && left[i - 1] === right[j - 1]) {
      temp.push({ left: left[i - 1], right: right[j - 1], type: 'same' });
      i--; j--;
    } else if (j > 0 && (i === 0 || dp[i][j - 1] >= dp[i - 1][j])) {
      temp.push({ left: '', right: right[j - 1], type: 'added' });
      j--;
    } else {
      temp.push({ left: left[i - 1], right: '', type: 'removed' });
      i--;
    }
  }
  temp.reverse();
 
  // Merge adjacent add/remove into changed
  let k = 0;
  while (k < temp.length) {
    if (temp[k].type === 'removed' && k + 1 < temp.length && temp[k + 1].type === 'added') {
      result.push({ left: temp[k].left, right: temp[k + 1].right, type: 'changed' });
      k += 2;
    } else {
      result.push(temp[k]);
      k++;
    }
  }
 
  return result;
}
 
function diffHtml(text: string): string {
  return text.replace(/&/g, '&amp;').replace(/</g, '&lt;').replace(/>/g, '&gt;');
}
 
function doDiff() {
  const leftLines = leftEl.value.split('\n');
  const rightLines = rightEl.value.split('\n');
  const diff = computeDiff(leftLines, rightLines);
 
  let added = 0, removed = 0, changed = 0;
  for (const d of diff) {
    if (d.type === 'added') added++;
    else if (d.type === 'removed') removed++;
    else if (d.type === 'changed') changed++;
  }
 
  statsEl.innerHTML = `
    共 <span>${diff.length}</span> 行 | 
    <span style="color:#f85149">删除 ${removed}</span> | 
    <span style="color:#3fb950">新增 ${added}</span> | 
    <span style="color:#d29922">修改 ${changed}</span> | 
    <span style="color:#8b949e">未变 ${diff.length - removed - added - changed}</span>
  `;
 
  let leftNum = 0, rightNum = 0;
  diffEl.innerHTML = diff.map((d, idx) => {
    if (d.left) leftNum++;
    if (d.right) rightNum++;
    const ln = d.left ? String(leftNum) : '';
    const rn = d.right ? String(rightNum) : '';
    return `<div class="diff-line ${d.type === 'same' ? '' : d.type}">
      <div class="diff-line-num">${ln}</div>
      <div class="diff-left">${diffHtml(d.left)}</div>
      <div class="diff-line-num">${rn}</div>
      <div class="diff-right">${diffHtml(d.right)}</div>
    </div>`;
  }).join('');
}
