function getTodayKey() { return new Date().toISOString().slice(0, 10); }
 
const STORAGE_DB = 'daily-sign-db';
const store = createIdbStore(STORAGE_DB, 'kv');
 
async function getSign(key: string): Promise<string | null> {
  return await store.get(key);
}
 
async function setSign(key: string, value: string): Promise<void> {
  await store.set(key, value);
}
 
async function drawSign(): Promise<{poem:number, hex:number[]}> {
  const today = getTodayKey();
  const key = `ds_${today}`;
  let result = await getSign(key);
  if (!result) {
    const idx = Math.floor(Math.random() * poems.length);
    const h1 = Math.floor(Math.random() * 8);
    const h2 = Math.floor(Math.random() * 8);
    result = JSON.stringify({ poem: idx, hex: [h1, h2] });
    await setSign(key, result);
  }
  return JSON.parse(result);
}

  const app = document.getElementById('app')!;
  app.innerHTML = `

    <div class="result">
      <div class="hexagram">${hexagrams[signData.hex[0]]}${hexagrams[signData.hex[1]]}</div>
      <div class="poem">${p.text}</div>
      <div class="fortune ${p.level}">「${p.fortune}」</div>

}
 
async function init() {
  const today = getTodayKey();
  const key = `ds_${today}`;
  const signed = await getSign(key);
 
  const app = document.getElementById('app')!;
  if (signed) {
    const signData = JSON.parse(signed);
    renderResult(signData);
  } else {
    app.innerHTML = `
